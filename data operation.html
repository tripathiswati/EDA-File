def EDA (A,figuresize):
    cat=[]
    con=[]
    import pandas as pd
    import matplotlib.pyplot as plt
    import seaborn as sb
    u=int((A.shape[1]+1)/2)
    plt.figure(figsize=figuresize)
    x=1
    for i in A.columns:
        if (A[i].dtype=="object"):
            cat.append(i)
            plt.subplot(u,2,x)
            x=x+1
            sb.countplot(A[i])
    else:
        cont.append(i)
        plt.subplot(u,2,x)
        x=x+1
        sb.countplot(A[i])
        print(A.corr())
